第一步：建立与数据库的连接

var request = window.indexedDB.open(name, versiono);
var db;
request.onerror = function(event){
}
request.onupgradeneeded = function(event){
    db = event.target.result;
}
request.onsuccess = function(event){
    db = event.target.result;
}

var store = db.createObjectStore("users", {keyPath: "ssn"});

使用游标查询(检索多个对象时)

var store = db.transaction("users").objectStore("users"),
    request = store.openCursor(); //创建游标
request.onerror = function(){
}
request.onsuccess = function(event){
    var cursor = event.target.result;  
    //通过event.target.result这个属性取得存储空间的下一个对象,有值时返回IDBCursor实例，无值返回null
    Cursor实例属性：
    cursor.direction //数值，表示游标移动的方向
    cursor.key //对象的键
    cursor.value //实际的对象，显示前JSON.stringify(cursor.value)
    cursor.primaryKey //游标使用的键
}

可以更新和删除个别记录

request.onsuccess = function(event){
    var cursor = event.target.result,
        value, updateRequest;

    if(cursor){
        if(cursor.key === "foo"){
            value = cursor.value;
            value.password = "magic";

            updateRequest = cursor.update(value); || deleteRequest = cursor.delete();//如果没有修改对象存储空间权限，这两个方法会抛出错误
            updateRequest.onerror = function(){
            }
            updateRequest.onsuccess = function(){
            }
        }
    }
}

每个游标只能发起一次请求，调用以下方法发起下一次请求

  • continue(key) //if(key){移动到key的位置}else{移动到结果集中的下一项}

  • advance(count) //向前移动count指定的项数

    request.onsuccess = function(event){

    var cursor = event.target.result;  
if(cursor){
    cursor.continue();
}else{
    console.log("done");
}

键范围(IDBKeyRange实例)

四种定义键范围的方式

  • var onlyRange = IDBKeyRange.only(key); //类似db.transaction(“users”, IDBTransaction.READ_WRITE).createStore(“users”).get(key);
  • var lowerRange = IDBKeyRange.lowerBound(key, boolean);
  • var upperRange = IDBKeyRange.upperBound(key, boolean);

  • var boundRange = IDBKeyRange.bound(key1, key2, boolean1, boolean2);

    var store = db.transaction(“users”).objectStore(“users”),

    range = IDBKeyRange.bound('ace', '007'),
request = store.openCursor(range);

    request.onsuccess = function(event){

    var cursor = event.target.result;  
if(cursor){
    cursor.continue();
}else{
    console.log("done");
}

    }

openCursor()的第二个参数表示设定游标方向

request = store.openCursor(range, IDBCursor.NEXT); //向后
request = store.openCursor(range, IDBCursor.NEXT_NO_DUPLICATE); //跳过重复的
request = store.openCursor(range, IDBCursor.PREV); //向前
request = store.openCursor(range, IDBCursor.PREV_NO_DUPLICATE); //跳过重复的

索引

var store = db.transaction("users").objectStore("users"),
    index = store.createIndex("username", "username", {unique: false}); //返回IDBIndex实例，等同于下面的方法

var store = db.transaction("users").objectStore("users"),
    index = store.index("username");
    //可以在索引上创建游标
    request = index.openCursor();
request.onsuccess = function(event){
    event.result.key //保存索引键
    event.result.value //保存主键
}

//也可以使用get方法取得一个对象
var store = db.transaction("users").objectStore("users"),
    index = store.index("username");
    request = index.get("007");
request.onsuccess = function(event){
}
request.onerror = function(event){
}

//可以使用getKey方法取得主键
var store = db.transaction("users").objectStore("users"),
    index = store.index("username");
    request = index.get("007");
request.onsuccess = function(event){
    event.result.key //保存索引键
    event.result.value //保存主键
}

任何时候可以通过IDBIndex对象取得有关索引的相关信息

  • name:索引名字
  • keyPath:传入createIndex()中的属性路径
  • objectStore:索引的对象存储空间
  • unique:表示索引键是否唯一的布尔值

通过对象存储对象的indexname属性可以访问为该空间建立的所有索引

var store = db.transaction("users").objectStore("users"),
    indexNames = store.indexNames;
    index,
    i = 0,
    len = indexNames.length;

while(i < len){
    index = store.index(indexNames[i++]);
    console.log(index.name);
    console.log(index.keyPath);
    console.log(index.objectStore);
    console.log(index.unique);
}

store.deleteIndex("username");
