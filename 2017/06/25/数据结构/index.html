function linkedList(){
    //表示要加入的项
    this.Node = function(element){
        this.element = element;
        this.next = null;
    };

    this.length = 0;
    this.head = null;
}

linkedList.prototype = {
    append: function(elem){
        var self = this,
            node = new self.Node(elem),
            current;

        if(self.head === null){//列表中第一个节点
            self.head = node;
        }else{
            current = self.head;

            //循环列表，直到找到最后一项
            while(current.next){
                current = current.next;
            }

            //找到最后一项，将其next赋为node，建立链接
            current.next = node;
        }

        self.length++;  //更新列表长度

    },
    insert: function(position, elem){
        var self = this;

        if(position > -1 && position < self.length){
            var node = new self.Node(elem),
                current = head,
                previous,
                index = 0;

            if(position === 0){
                node.next = self.head;
                self.head = node;
            }else{
                while(index++ < position){
                    previous = current;
                    current = current.next;
                }
                node.next = current;
                previous.next = node;
            }

            self.length++;

            return true;
        }else{
            return false;
        }
    },
    removeAt: function(position){
        var self = this;

        if(position > -1 && position < self.length){
            var current = self.head,
                previous,
                index = 0;

             if(position === 0){    //移除第一项
                self.head = current.next;
             }else{
                while(index++ < position){
                    previous = current;
                    current = current.next;
                }

                //将previous与current的下一项连接起来，跳过current，从而移除它
                previous.next = current.next;
             }

             self.length--;

             return current.element;
        }else{
            return null;
        }
    },
    remove: function(elem){
        var self = this;

        var index = self.indexOf(elem);
        return self.removeAt(index);
    },
    indexOf: function(elem){
        var self = this;

        var current = self.head,
            index = 0;

        while(current){
            if(elem === current.element){
                return index;
            }
            index++;
            current = current.next;
        }

        return -1;
    },
    isEmpty: function(){
        var self = this;

        return self.length === 0;
    },
    size: function(){
        var self = this;

        return self.length;
    },
    getHead: function(){
        var self = this;

        return self.head;
    },
    toString: function(){
        var self = this;

        var current = self.head,
            string = '';

        while(current){
            string += ',' + current.element;
            current = current.next;
        }

        return string.slice(1);     
    },
    print: function(){
        var self = this;
    }
}

function HashTable(){
    var table = [];
    var loselloseHashCode = function(key){
        var hash = 0;
        for(var i=0;i<key.length;i++){
            hash += key.charCodeAt(i);
        }
        return hash % 37;
    }

    this.put = function(key, value){
        var position = loselloseHashCode(key);
        console.log(position + '-' + key);
        table[position] = value;
    };

    this.get = function(key){
        return table[loselloseHashCode(key)];
    };

    this.remove = function(key){
        table[loselloseHashCode(key)] = undefined;
    };

    this.print = function(){
        for(var i=0;i<table.length;i++){
            if(table[i] !== undefined){
                console.log(i + ': ' + table[i]);
            }
        }
    }
}
